===
this
---
`this` 要在执行时才能确认值，定义时无法确认。

作为构造函数执行
```js
function Foo(name){
	this.name = name
}
var f = new Foo('zhangsan')
```
作为对象属性执行（`this`是这个对象）
```js
var obj = {
	name:'A',
	printName: function (){
		console.log(this.name)
	}
}
obj.printName()
```
作为普通的函数执行
```js
function fn(){
	console.log(this)//this === window
}
fn()
```
call apply bind
```js
const name = 'xiaoming';

const a = function (name){

  console.log(this.name);

  console.log(this);

}

a.call({name:'xiaohong'});//xiaohong
```

闭包的作用
---
1. 函数作为返回值
2. 函数作为参数传递

```js
function  F1(){
	var  a  =  100
	return  function  (){
		console.log(a)//自由变量，父作用域寻找
	}
}

var  f1  =  F1()
var  a  =  200

f1()//100

function  F2(fn){
	var  a  =  300;
	fn();
}

F2(f1);//100
```
